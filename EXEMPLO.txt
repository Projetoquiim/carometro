Cenário A: Rotas no Frontend (Sem usar Express)Use este se você quer apenas trocar de página no navegador (ex: / para /contato) sem precisar de um servidor processando os dados.

1. Estrutura de arquivos:textseu-projeto/
├── public/
│   ├── index.html
│   └── script.js
└── firebase.json
Use o código com cuidado.

2. Código do public/index.html:html<!DOCTYPE html>
<html lang="pt-BR">
<head>
    <meta charset="UTF-8">
    <title>Meu App Estático</title>
</head>
<body>
    <nav>
        <button onclick="navegar('/')">Home</button>
        <button onclick="navegar('/contato')">Contato</button>
    </nav>

    <!-- As páginas serão injetadas aqui dentro -->
    <div id="conteudo"></div>

    <script src="/script.js"></script>
</body>
</html>
Use o código com cuidado.

3. Código do public/script.js (O roteador do navegador):

javascript// Define o que aparece em cada rota
const rotas = {
    '/': '<h1>Página Inicial</h1><p>Bem-vindo ao meu site!</p>',
    '/contato': '<h1>Contato</h1><p>Fale conosco em contato@email.com</p>',
    '/404': '<h1>404</h1><p>Página não encontrada.</p>'
};

// Função para mudar a URL e o conteúdo sem recarregar a página
function navegar(url) {
    window.history.pushState({}, "", url);
    processarRota();
}

// Renderiza o conteúdo baseado na URL atual
function processarRota() {
    const caminho = window.location.pathname;
    const conteudoDiv = document.getElementById('conteudo');
    conteudoDiv.innerHTML = rotas[caminho] || rotas['/404'];
}

// Escuta os botões de voltar/avançar do navegador
window.onpopstate = processarRota;

// Processa a rota assim que a página carrega
processarRota();

  "hosting": {
    "public": "public",
    "ignore": [
      "firebase.json",
      "**/.*",
      "**/node_modules/**"
    ],
    "rewrites": [
      {
        "source": "**",
        "destination": "/index.html"
      }
    ]
  }
}
